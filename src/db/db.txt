CREATE TABLE languages (
  id SERIAL PRIMARY KEY,
  code TEXT UNIQUE,   -- en, es
  label TEXT          -- English, Spanish
);

-- Sequence and defined type
CREATE SEQUENCE IF NOT EXISTS users_id_seq;

-- Table Definition
CREATE TABLE "public"."users" (
    "id" int4 NOT NULL DEFAULT nextval('users_id_seq'::regclass),
    "email" text NOT NULL,
    "password" text NOT NULL,
    "created_at" timestamp DEFAULT CURRENT_TIMESTAMP,
    "username" text,
    "language_code" text,
    CONSTRAINT "users_language_code_fkey" FOREIGN KEY ("language_code") REFERENCES "public"."languages"("code"),
    PRIMARY KEY ("id")
);

-- Sequence and defined type
CREATE SEQUENCE IF NOT EXISTS languages_id_seq;

-- Table Definition
CREATE TABLE "public"."languages" (
    "id" int4 NOT NULL DEFAULT nextval('languages_id_seq'::regclass),
    "code" text,
    "label" text,
    PRIMARY KEY ("id")
);

CREATE TABLE lessons (
    id SERIAL PRIMARY KEY,
    title TEXT NOT NULL,
    "order" INT NOT NULL,
    locked BOOLEAN DEFAULT FALSE,

    language_code TEXT REFERENCES languages(code)
);

CREATE TABLE steps (
    id SERIAL PRIMARY KEY,
    lesson_id INT REFERENCES lessons(id) ON DELETE CASCADE,

    type TEXT CHECK (type IN ('flashcard', 'quiz')),

    word TEXT,
    translation TEXT,

    question TEXT,
    correct_answer TEXT,

    "order" INT NOT NULL
);

CREATE TABLE quiz_options (
    id SERIAL PRIMARY KEY,
    step_id INT REFERENCES steps(id) ON DELETE CASCADE,
    text TEXT NOT NULL
);

CREATE TABLE user_progress (
  id SERIAL PRIMARY KEY,
  user_id INT REFERENCES users(id) ON DELETE CASCADE,
  lesson_id INT REFERENCES lessons(id) ON DELETE CASCADE,
  progress INT DEFAULT 0,
  completed BOOLEAN DEFAULT FALSE,

  UNIQUE(user_id, lesson_id)
);

CREATE TABLE user_stats (
  id SERIAL PRIMARY KEY,
  user_id INT UNIQUE REFERENCES users(id) ON DELETE CASCADE,
  xp INT DEFAULT 0,
  streak INT DEFAULT 0
);

CREATE UNIQUE INDEX lessons_order_unique
ON lessons(language_code, "order");

CREATE UNIQUE INDEX steps_order_unique
ON steps(lesson_id, "order");
